function suffix_in(host,list)
{
	var pos=host.length;
	while(pos>0)
	{
		pos=host.lastIndexOf(".",pos-1);
		var suffix=host.substring(pos+1);
		if (Object.hasOwnProperty.call(list,suffix)) return true;
	}
	return false;
}

function re_find(url,list_re)
{
	return list_re.some(function (item_re){ return item_re.test(url);});
}

function FindProxyForURL(url,host) {
	if (suffix_in(host,exceptions)) return direct;
	if (re_find(url,exception_re)) return direct;
	if (suffix_in(host,domains)) return proxy;
	if (re_find(url,rule_re)) return proxy;
	return direct;
}
